        const parsePastedData = (text) => {

            let matrix = [];

            if (text.indexOf('\t') !== -1) {

                matrix = text.split('\n').map(row => row.split('\t').map(c => {

                    let cell = c.trim();

                    if (cell.startsWith('"') && cell.endsWith('"')) return cell.substring(1, cell.length - 1);

                    return cell;

                }));

            } else {

                let row = [], field = '', inQuotes = false;

                for (let i = 0; i < text.length; i++) {

                    let char = text[i];

                    if (char === '"') {

                        if (inQuotes && text[i+1] === '"') { field += '"'; i++; } else { inQuotes = !inQuotes; }

                    } else if (char === ',' && !inQuotes) { row.push(field); field = ''; } 

                    else if ((char === '\n' || (char === '\r' && text[i+1] === '\n')) && !inQuotes) {

                        if (char === '\r') i++;

                        row.push(field); matrix.push(row); row = []; field = '';

                    } else { field += char; }

                }

                if (field || row.length > 0) { row.push(field); matrix.push(row); }

            }

            return matrix;

        };


        const processExcelData = (matrix) => {

            if (!matrix || matrix.length < 2) return [];

            const headers = matrix[0].map(h => h.trim().replace(/^"|"$/g, ''));

            const colMap = {

                category: Math.max(0, headers.findIndex(h => h.includes('分类') || h.includes('板块'))),

                no: Math.max(1, headers.findIndex(h => h.includes('序号'))),

                task: Math.max(2, headers.findIndex(h => h.includes('事项') || h.includes('项目'))),

                detail: Math.max(3, headers.findIndex(h => h.includes('操作') || h.includes('详细') || h.includes('备注'))),

                progress: Math.max(4, headers.findIndex(h => h.includes('进展') || h.includes('预期'))),

                status: Math.max(5, headers.findIndex(h => h.includes('状态'))),

                assignee: Math.max(6, headers.findIndex(h => h.includes('参与人') || h.includes('负责人'))),

            };


            let data = [], lastCat = "", lastNo = "", lastTask = "";

            for (let i = 1; i < matrix.length; i++) {

                const row = matrix[i];

                if (!row || row.length === 0 || row.every(c => !c || c.trim() === '')) continue;

                

                const getCell = (idx) => (idx === -1 || idx >= row.length) ? "" : (row[idx] || "").trim().replace(/^"|"$/g, '');


                let cat = getCell(colMap.category); if (cat) lastCat = cat; else cat = lastCat;

                let no = getCell(colMap.no); if (no) lastNo = no; else no = lastNo;

                let task = getCell(colMap.task); if (task) lastTask = task; else task = lastTask;

                

                let detail = getCell(colMap.detail), progress = getCell(colMap.progress);

                let statusRaw = getCell(colMap.status) || "To Do", assignee = getCell(colMap.assignee);

                if (!detail && !progress && !assignee && (!statusRaw || statusRaw === "To Do")) continue;


                let status = "To Do";

                const sLower = statusRaw.toLowerCase();

                if (sLower.includes("progress") || sLower.includes("进行")) status = "In progress";

                else if (sLower.includes("done") || sLower.includes("完成")) status = "Done";

                else if (sLower.includes("pending") || sLower.includes("暂缓")) status = "Pending";

                else if (statusRaw.trim()) status = statusRaw; 

                

                data.push({ id: `task-imp-${Date.now()}-${i}`, category: cat || "默认板块", no, task: task || "未命名事项", detail, progress, status, assignee: assignee || "未分配", updatedAt: Date.now() });

            }

            return data;

        };

                                                <th className="px-4 py-3 text-right text-xs font-medium text-slate-500 uppercase tracking-wider w-24">最后更新</th>

                                            </tr>

                                        </thead>

                                        <tbody className="bg-white divide-y divide-slate-100">

                                            {filteredTasks.map(task => (

                                                <tr key={task.id} onClick={() => setEditingTask(task)} className="hover:bg-indigo-50/50 transition-colors cursor-pointer group">

                                                    <td className="px-4 py-4 whitespace-nowrap text-sm text-slate-600"><span className="bg-slate-100 px-2 py-1 rounded text-xs">{task.category}</span></td>

                                                    <td className="px-4 py-4 text-sm text-slate-900 font-medium">{task.task} {task.no && <span className="ml-2 text-xs text-slate-400 font-normal">#{task.no}</span>}</td>

                                                    <td className="px-4 py-4 text-sm text-slate-600"><div className="line-clamp-2 whitespace-pre-wrap">{task.progress || <span className="text-slate-300 italic">无记录</span>}</div></td>

                                                    <td className="px-4 py-4 whitespace-nowrap">

                                                        <span className={`px-2.5 py-1 inline-flex text-[10px] leading-5 font-semibold rounded-full border ${task.status === 'Done' ? 'bg-emerald-50 text-emerald-700 border-emerald-200' : task.status === 'In progress' ? 'bg-indigo-50 text-indigo-700 border-indigo-200' : 'bg-slate-50 text-slate-700 border-slate-200'}`}>

                                                            {task.status}

                                                        </span>

                                                    </td>

                                                    <td className="px-4 py-4 whitespace-nowrap"><Avatar name={task.assignee} /></td>

                                                    <td className="px-4 py-4 whitespace-nowrap text-right text-xs text-slate-400">{timeAgo(task.updatedAt)}</td>

                                                </tr>

                                            ))}

                                            {filteredTasks.length === 0 && <tr><td colSpan="6" className="px-6 py-12 text-center text-slate-400">当前视图无数据</td></tr>}

                                        </tbody>

                                    </table>

                                </div>

                            </div>

                        )}

                    </main>


                    {/* 模态框：编辑/新增卡片 */}

                    {editingTask && (

                        <div className="fixed inset-0 bg-slate-900/40 backdrop-blur-sm z-50 flex items-center justify-center p-4 modal-overlay">

                            <div className="bg-white rounded-2xl shadow-2xl w-full max-w-2xl overflow-hidden flex flex-col max-h-[90vh] modal-content border border-slate-100">

                                <div className="px-6 py-4 border-b border-slate-100 flex justify-between items-center bg-slate-50">

                                    <h2 className="text-lg font-bold text-slate-800 flex items-center">

                                        {editingTask.id ? (isOffline ? '编辑本地任务' : '多人协作 · 实时编辑') : '创建新任务'}

                                        {saving && <span className="ml-3 text-xs text-indigo-500 font-normal flex items-center">...正在云端同步</span>}

                                    </h2>

                                    <button type="button" onClick={() => setEditingTask(null)} className="text-slate-400 hover:text-slate-700 transition-colors p-1" disabled={saving}>✕</button>

                                </div>

                                

                                <form id="task-form" onSubmit={handleSaveTask} className="flex-1 overflow-y-auto p-6 space-y-5">

                                    <div className="grid grid-cols-2 gap-5">

                                        <div><label className="block text-xs font-medium text-slate-500 mb-1">板块 / 分类</label><input name="category" defaultValue={editingTask.category} className="w-full px-3 py-2 border border-slate-300 rounded-lg focus:ring-2 focus:ring-indigo-500 text-sm" placeholder="例如：顶层设计" disabled={saving} /></div>

                                        <div><label className="block text-xs font-medium text-slate-500 mb-1">序号 (可选)</label><input name="no" defaultValue={editingTask.no} className="w-full px-3 py-2 border border-slate-300 rounded-lg focus:ring-2 focus:ring-indigo-500 text-sm" disabled={saving} /></div>

                                    </div>

                                    

                                    <div><label className="block text-xs font-medium text-slate-500 mb-1">重点事项标题 <span className="text-red-500">*</span></label><input required name="task" defaultValue={editingTask.task} className="w-full px-3 py-2 border border-slate-300 rounded-lg focus:ring-2 focus:ring-indigo-500 text-sm font-medium text-slate-900" disabled={saving} /></div>

                                    <div><label className="block text-xs font-medium text-slate-500 mb-1">进展情况及预期 (将高亮展示在团队看板上)</label><textarea name="progress" defaultValue={editingTask.progress} rows="3" className="w-full px-3 py-2 border border-slate-300 rounded-lg focus:ring-2 focus:ring-indigo-500 text-sm whitespace-pre-wrap leading-relaxed" disabled={saving}></textarea></div>

                                    <div><label className="block text-xs font-medium text-slate-500 mb-1">具体操作 / 详细说明 (长文本备注)</label><textarea name="detail" defaultValue={editingTask.detail} rows="3" className="w-full px-3 py-2 border border-slate-300 rounded-lg focus:ring-2 focus:ring-indigo-500 text-sm whitespace-pre-wrap leading-relaxed bg-slate-50" disabled={saving}></textarea></div>


                                    <div className="grid grid-cols-2 gap-5 bg-slate-50 p-4 rounded-xl border border-slate-100">

                                        <div>

                                            <label className="block text-xs font-medium text-slate-500 mb-1">当前状态</label>

                                            <select name="status" defaultValue={editingTask.status || 'To Do'} className="w-full px-3 py-2 border border-slate-300 rounded-lg focus:ring-2 focus:ring-indigo-500 text-sm bg-white" disabled={saving}>

                                                <option value="To Do">待处理 (To Do)</option>

                                                <option value="In progress">进行中 (In Progress)</option>

                                                <option value="Pending">等待/暂缓 (Pending)</option>

                                                <option value="Done">已完成 (Done)</option>

                                            </select>

                                        </div>

                                        <div><label className="block text-xs font-medium text-slate-500 mb-1">主要参与人 (逗号分隔)</label><input name="assignee" defaultValue={editingTask.assignee} className="w-full px-3 py-2 border border-slate-300 rounded-lg focus:ring-2 focus:ring-indigo-500 text-sm bg-white" placeholder="如：Finn, Lydia" disabled={saving} /></div>

                                    </div>

                                </form>

                                

                                <div className="px-6 py-4 bg-slate-50 border-t border-slate-100 flex justify-between items-center">

                                    {editingTask.id ? <button type="button" onClick={handleDeleteTask} disabled={saving} className="text-red-500 hover:text-red-700 hover:bg-red-50 px-3 py-1.5 rounded-lg text-sm transition-colors">彻底删除</button> : <div></div>}

                                    <div className="flex space-x-3">

                                        <button type="button" onClick={() => setEditingTask(null)} disabled={saving} className="px-5 py-2 text-sm text-slate-600 hover:bg-slate-200 rounded-lg font-medium">取消</button>

                                        <button type="submit" form="task-form" disabled={saving} className="px-6 py-2 text-sm font-bold text-white bg-indigo-600 hover:bg-indigo-700 rounded-lg shadow-sm">

                                            {saving ? '保存中...' : (editingTask.id ? (isOffline ? '保存本地修改' : '✨ 立即同步至团队') : '确认创建')}

                                        </button>

                                    </div>

                                </div>

                            </div>

                        </div>

                    )}


                    {/* 模态框：导入 Excel */}

                    {isImportModalOpen && (

                        <div className="fixed inset-0 bg-slate-900/50 backdrop-blur-sm z-50 flex items-center justify-center p-4 modal-overlay">

                            <div className="bg-white rounded-2xl shadow-xl w-full max-w-3xl overflow-hidden flex flex-col max-h-[90vh] modal-content">

                                <div className="px-6 py-4 border-b border-slate-100 flex justify-between items-center bg-slate-50">

                                    <h2 className="text-lg font-bold text-slate-800">📊 团队进度批量导入</h2>

                                    <button onClick={() => setIsImportModalOpen(false)} className="text-slate-400 hover:text-slate-600" disabled={saving}>✕</button>

                                </div>

                                

                                <div className="p-6 flex-1 overflow-y-auto">

                                    <div className="bg-blue-50 text-blue-800 text-sm p-4 rounded-xl mb-4 border border-blue-100">

                                        <p className="font-bold mb-1">如何从 Excel 导入？</p>

                                        <ol className="list-decimal pl-5 space-y-1 text-blue-700">

                                            <li>在您的 Excel/WPS 中框选需要导入的行（<b>需包含表头</b>）。</li>

                                            <li>按 <code>Ctrl+C</code> 复制。</li>

                                            <li>在下方文本框按 <code>Ctrl+V</code> 粘贴，点击确认。系统会自动解析并在云端为您追加这些任务！</li>

                                        </ol>

                                    </div>

                                    <textarea 

                                        className="w-full h-48 p-4 border border-slate-300 rounded-xl focus:ring-2 focus:ring-indigo-500 font-mono text-sm whitespace-pre"

                                        placeholder="请在此处粘贴 Excel 内容..." value={importText} onChange={(e) => setImportText(e.target.value)} disabled={saving}

                                    ></textarea>

                                </div>

                                

                                <div className="px-6 py-4 bg-slate-50 border-t border-slate-100 flex justify-end space-x-3">

                                    <button onClick={() => setIsImportModalOpen(false)} className="px-5 py-2 text-sm font-medium text-slate-600 hover:bg-slate-200 rounded-lg" disabled={saving}>取消</button>

                                    <button onClick={handleImportSubmit} className="px-5 py-2 text-sm font-bold text-white bg-indigo-600 hover:bg-indigo-700 rounded-lg shadow-sm" disabled={saving}>

                                        {saving ? '正在同步至云端...' : '🚀 解析并同步全团队'}

                                    </button>

                                </div>

                            </div>

                        </div>

                    )}


                    {/* 全局 Toast */}

                    {toastMsg && (

                        <div className="fixed bottom-6 right-6 bg-slate-800 text-white px-5 py-3 rounded-xl shadow-lg flex items-center space-x-3 animate-bounce z-50">

                            <span className="text-xl">✨</span><span className="text-sm font-medium">{toastMsg}</span>

                        </div>

                    )}

                </div>

            );

        };


        const root = ReactDOM.createRoot(document.getElementById('root'));

        root.render(<App />);

    </script>
